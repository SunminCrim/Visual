# Sample data generation (replace with your actual data)
set.seed(123)
data <- data.frame(
  x = rnorm(100),
  y = rnorm(100),
  size = abs(rnorm(100)) * 10  # for bubble chart
)

# UI Definition
ui <- fluidPage(
    titlePanel("Interactive Data Visualization Dashboard"),
    
    sidebarLayout(
        sidebarPanel(
            selectInput("chart_type", "Select Chart:",
                       choices = c("Scatter Plot (Chart 9)", 
                                 "Bubble Chart (Chart 18)",
                                 "Additional Chart")),
            # Interactive controls
            sliderInput("point_size", "Point Size:",
                       min = 1, max = 10, value = 3),
            selectInput("color_scheme", "Color Theme:",
                       choices = c("Default", "Blue Scale", "Red Scale"))
        ),
        
        mainPanel(
            plotlyOutput("plot")
        )
    )
)

# Server Logic
server <- function(input, output) {
    output$plot <- renderPlotly({
        p <- if(input$chart_type == "Scatter Plot (Chart 9)") {
            # Scatter plot implementation
            ggplot(data, aes(x = x, y = y)) +
                geom_point(size = input$point_size) +
                labs(title = "Scatter Plot",
                     x = "X Variable",
                     y = "Y Variable") +
                theme_minimal()
            
        } else if(input$chart_type == "Bubble Chart (Chart 18)") {
            # Bubble chart implementation
            ggplot(data, aes(x = x, y = y, size = size)) +
                geom_point(alpha = 0.6) +
                labs(title = "Bubble Chart",
                     x = "X Variable",
                     y = "Y Variable",
                     size = "Size") +
                theme_minimal()
        } else {
            # Additional chart implementation
            ggplot(data, aes(x = x)) +
                geom_histogram(binwidth = 0.5) +
                labs(title = "Histogram",
                     x = "Value",
                     y = "Frequency") +
                theme_minimal()
        }
        
        # Apply color theme
        if(input$color_scheme == "Blue Scale") {
            p <- p + scale_color_brewer(palette = "Blues")
        } else if(input$color_scheme == "Red Scale") {
            p <- p + scale_color_brewer(palette = "Reds")
        }
        
        ggplotly(p)  # Convert to interactive plot
    })
}

# Run the application
shinyApp(ui = ui, server = server)
